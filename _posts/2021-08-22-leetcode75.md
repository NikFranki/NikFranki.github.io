### 代码

```js
var sortColors = function(nums) {
    // 计数排序
    if (nums.length === 0) {
        return;
    }
    let r = 0, w = 0, b = 0;
    for (const num of nums) {
        if (num === 0) {
            nums[b] = 2;
            b++;
            nums[w] = 1;
            w++;
            nums[r] = 0;
            r++; 
        } else if (num === 1) {
            nums[b] = 2;
            b++;
            nums[w] = 1;
            w++;
        } else if (num === 2) {
            b++;
        }
    }
};
```
